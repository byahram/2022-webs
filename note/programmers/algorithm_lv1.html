01. 자연수 뒤집어 배열로 만들기 @221005

자연수 n을 뒤집어 각 자리 숫자를 원소로 가지는 배열 형태로 리턴해주세요. 예를들어 n이 12345이면 [5,4,3,2,1]을 리턴합니다.

// 나의 풀이
function solution(n) {
    var answer=[];
    var reverseArr = String(n).split('').reverse();
    
    for(let i=0; i < reverseArr.length; i++) {
        answer[i]=Number(reverseArr[i]);
    }
    return answer;
}
    
    
// 다른 사람의 풀이
function solution(n) {
    return n.toString().split('').reverse().map(o => o = parseInt(o));
}

02. 평균 구하기 @221005

정수를 담고 있는 배열 arr의 평균값을 return하는 함수, solution을 완성해보세요.

// 나의 풀이
function solution(arr) {
    var sum = 0, avg = 0;
    for(let i = 0; i < arr.length; i++) {
        sum += arr[i];
    }
    avg = sum / arr.length;
    return avg;
}
    
    
// 다른 사람이 제출한 es6식 문법
function average(array){
  return array.reduce((a, b) => a + b) / array.length;
}

// 아래는 테스트로 출력해 보기 위한 코드입니다.
var testArray = [5,3,4] 
console.log("평균값 : " + average(testArray));

03. 가운데 글자 가져오기 @221026

단어 s의 가운데 글자를 반환하는 함수, solution을 만들어 보세요. 단어의 길이가 짝수라면 가운데 두글자를 반환하면 됩니다.

// 나의 풀이
function solution(s) {
    var answer = '';
    
    if( s.length % 2 == 0) {
        answer = s.substring((s.length/2)-1, (s.length/2)+1);
    } else {
        answer = s.substring((s.length/2), (s.length/2)+1);
    }
    return answer;
}
    
    
// 다른 사람이 제출한 es6식 문법
// 1. 
function solution(s) {
    return s.substr(Math.ceil(s.length / 2) - 1, s.length % 2 === 0 ? 2 : 1);
}

// 2.
function solution(s) {
    const mid = Math.floor(s.length/2);
    return s.length %2 === 1 ? s[mid] : s[mid-1]+s[mid];
}

04. 약수의 합 @221028

정수 n을 입력받아 n의 약수를 모두 더한 값을 리턴하는 함수, solution을 완성해주세요.

// 나의 풀이
function solution(n) {
    var answer = 0;
    
    for(let i = 0; i <= n; i++) {
        if(n % i === 0) {
            answer = answer + i;
        }
    }
    return answer;
}
    
    
// 다른 사람의 풀이
// 1. 
function solution(num) {
    let sum = 0;
    for (let i = 1; i <= num; i++) {
        if (num % i === 0) sum += i
    }
    return sum
}

//2. 재귀+삼항...
function solution(n, a=0, b=0) {
    return n <= a/2?b:solution(n, a+1, b+=n%a?0:a);
}

05. 자릿수 더하기 @221028

자연수 N이 주어지면, N의 각 자릿수의 합을 구해서 return 하는 solution 함수를 만들어 주세요. 예를들어 N = 123이면 1 + 2 + 3 = 6을 return 하면 됩니다.

// 제출답 : 숫자를 string으로 변환하여 푸는 방법
function solution(n) {
    var answer = 0;    
    var strArray = n.toString().split('');
    
    for(var i = 0; i < strArray.length; i++) {        
        answer += parseInt(strArray[i]);    
    }
    return answer;
}
    
    
// 다른 사람의 풀이
// 1.
function solution(n) {
    return (n+"").split("").reduce((acc, curr) => acc + parseInt(curr), 0)
}

// 2. 10으로 나누며 나머지를 더하는 방법
function solution(n) {
    var sum = 0;
    var ori_n =n;
    for(var i = 0; i < ori_n.toString().length; i++){
        sum +=  n%10;
        n = parseInt(n/10);
    }
    return sum;
}
